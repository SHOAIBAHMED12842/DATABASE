CREATE OR REPLACE TRIGGER QTYIN_049
BEFORE INSERT ON HASSAN_AHMED_SE
FOR EACH ROW
DECLARE
  Q NUMBER(8);
BEGIN
  SELECT ITME_QTY INTO Q FROM HASSAN_AHMED WHERE ITEM_ID=:new.ITEM_ID;
  IF :new.ORDER_QTY > Q
  THEN
    RAISE_APPLICATION_ERROR(-20001,'ORDER QUANTITY GREATER THAN ITEM QUANTITY');
  end if;
END;
